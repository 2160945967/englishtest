        // 有道API工具函数
        const YoudaoAPI = {
            // 检查API配置是否完整
            isConfigured() {
                return !!youdaoConfig.appKey && !!youdaoConfig.appSecret;
            },
            
            // 生成签名
            generateSign(q, salt, appKey, appSecret) {
                return CryptoJS.MD5(appKey + q + salt + appSecret).toString();
            },
            
            // 获取单词详情
            async getWordDetail(word) {
                if (!this.isConfigured()) {
                    elements.apiError.style.display = "block";
                    return null;
                }
                
                const q = word.trim();
                const salt = new Date().getTime();
                const sign = this.generateSign(q, salt, youdaoConfig.appKey, youdaoConfig.appSecret);
                
                try {
                    const response = await fetch(`https://openapi.youdao.com/api`, {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/x-www-form-urlencoded'
                        },
                        body: new URLSearchParams({
                            q: q,
                            from: 'EN',
                            to: 'zh-CHS',
                            appKey: youdaoConfig.appKey,
                            salt: salt.toString(),
                            sign: sign,
                            signType: 'v3',
                            curtime: Math.round(new Date().getTime() / 1000).toString()
                        })
                    });
                    
                    const data = await response.json();
                    
                    // 处理有道API返回结果
                    if (data.errorCode !== '0') {
                        console.error('有道API错误:', data.errorCode, this.getErrorMsg(data.errorCode));
                        elements.meaningError.textContent = `API错误: ${this.getErrorMsg(data.errorCode)}`;
                        elements.meaningError.style.display = "block";
                        return null;
                    }
                    
                    return this.parseWordData(data, word);
                } catch (error) {
                    console.error('获取单词信息失败:', error);
                    elements.meaningError.textContent = "网络错误，无法获取单词信息";
                    elements.meaningError.style.display = "block";
                    return null;
                }
            },
            
            // 解析有道API返回的数据
            parseWordData(data, originalWord) {
                const result = {
                    word: originalWord,
                    phonetic: '',
                    parts: [],
                    examples: []
                };
                
                // 提取音标
                if (data.basic && data.basic.phonetic) {
                    result.phonetic = `/${data.basic.phonetic}/`;
                }
                
                // 提取释义
                if (data.basic && data.basic.explains) {
                    data.basic.explains.forEach(explain => {
                        // 简单解析词性和释义
                        const posMatch = explain.match(/^([a-zA-Z.]+)\s+(.*)/);
                        if (posMatch) {
                            const pos = this.convertPos(posMatch[1]);
                            const meaning = posMatch[2];
                            result.parts.push({ pos, meaning });
                        } else {
                            result.parts.push({ pos: '', meaning: explain });
                        }
                    });
                }
                
                // 提取例句
                if (data.web) {
                    data.web.forEach(item => {
                        if (item.value && item.value.length > 0) {
                            result.examples.push({
                                en: item.key,
                                cn: item.value.join('；')
                            });
                        }
                    });
                }
                
                return result;
            },
            
            // 转换词性缩写为中文显示
            convertPos(pos) {
                const posMap = {
                    'n.': 'n.（名词）',
                    'v.': 'v.（动词）',
                    'adj.': 'adj.（形容词）',
                    'adv.': 'adv.（副词）',
                    'prep.': 'prep.（介词）',
                    'conj.': 'conj.（连词）',
                    'pron.': 'pron.（代词）',
                    'num.': 'num.（数词）',
                    'int.': 'int.（感叹词）'
                };
                return posMap[pos] || pos;
            },
            
            // 获取错误信息
            getErrorMsg(errorCode) {
                const errorMap = {
                    '101': '缺少必填的参数',
                    '102': '不支持的语言类型',
                    '103': '翻译文本过长',
                    '104': '不支持的API类型',
                    '105': '不支持的签名类型',
                    '106': '不支持的响应格式',
                    '107': '不支持的加密类型',
                    '108': 'appKey无效',
                    '109': 'batchLog格式不正确',
                    '110': '无相关服务的有效实例',
                    '111': '开发者账号无效',
                    '112': '请求服务无效',
                    '113': 'q不能为空',
                    '201': '解密失败',
                    '202': '签名检验失败',
                    '203': '访问IP地址不在可访问IP列表中',
                    '204': '请求的接口与应用的平台类型不一致',
                    '301': '辞典查询失败',
                    '302': '翻译查询失败',
                    '303': '服务端的其它异常',
                    '401': '账户已经欠费',
                    '402': 'offline参数有误'
                };
                return errorMap[errorCode] || `未知错误(${errorCode})`;
            },
            
            // 获取单词发音URL
            getPronunciationUrl(word) {
                if (!this.isConfigured()) return null;
                // 有道发音URL格式
                return `https://dict.youdao.com/dictvoice?type=2&audio=${encodeURIComponent(word)}`;
            }

        // 数据管理模块
        const DataManager = {
            load: () => {
                try {
                    const savedLibraries = localStorage.getItem("wordLibraries");
                    const savedWrongWords = localStorage.getItem("wrongWordList");
                    const savedStats = localStorage.getItem("practiceStats");
                    
                    if (savedLibraries) {
                        wordLibraries = JSON.parse(savedLibraries);
                    }
                    
                    if (savedWrongWords) {
                        wrongWordList = JSON.parse(savedWrongWords);
                        Renderer.renderWrongWords();
                    }
                    
                    if (savedStats) {
                        const stats = JSON.parse(savedStats);
                        totalCount = stats.totalCount || 0;
                        correctCount = stats.correctCount || 0;
                    }
                } catch (e) {
                    console.error("加载数据失败:", e);
                    alert("加载保存的数据失败，将使用默认数据");
                }
            },
            
            save: () => {
                try {
                    localStorage.setItem("wordLibraries", JSON.stringify(wordLibraries));
                    localStorage.setItem("wrongWordList", JSON.stringify(wrongWordList));
                    localStorage.setItem("practiceStats", JSON.stringify({
                        totalCount,
                        correctCount
                    }));
                } catch (e) {
                    console.error("保存数据失败:", e);
                    alert("数据保存失败，可能是存储空间不足。建议导出备份。");
                }
            },
            
            addWord: (word) => {
                if (!wordLibraries[currentLibraryId].words) {
                    wordLibraries[currentLibraryId].words = [];
                }
                wordLibraries[currentLibraryId].words.push(word);
                DataManager.save();
            },
            
            deleteWord: (wordId) => {
                if (wordLibraries[currentLibraryId].words) {
                    wordLibraries[currentLibraryId].words = wordLibraries[currentLibraryId].words
                        .filter(word => word.id !== wordId);
                    DataManager.save();
                }
            },
            
            updateWordStats: (wordId, isCorrect) => {
                const library = wordLibraries[currentLibraryId];
                if (library && library.words) {
                    const wordIndex = library.words.findIndex(word => word.id === wordId);
                    if (wordIndex !== -1) {
                        if (isCorrect) {
                            library.words[wordIndex].correctCount = (library.words[wordIndex].correctCount || 0) + 1;
                        } else {
                            library.words[wordIndex].errorCount = (library.words[wordIndex].errorCount || 0) + 1;
                        }
                        DataManager.save();
                    }
                }
            }
        };

        // 初始化单词池
        function initWordPool() {
            const library = wordLibraries[currentLibraryId];
            if (library && library.words && library.words.length > 0) {
                // 复制单词数组并打乱顺序
                shuffledWordPool = [...library.words];
                shuffleArray(shuffledWordPool);
            } else {
                shuffledWordPool = [];
            }
        }

        // 打乱数组顺序
        function shuffleArray(array) {
            for (let i = array.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [array[i], array[j]] = [array[j], array[i]];
            }
        }

        // 获取下一个单词
        function getNextWord() {
            if (isRedoWrongMode && wrongWordList.length > 0) {
                // 重练错题模式
                const randomIndex = Math.floor(Math.random() * wrongWordList.length);
                currentWord = wrongWordList[randomIndex];
            } else {
                // 正常模式
                if (shuffledWordPool.length === 0) {
                    initWordPool();
                }
                currentWord = shuffledWordPool.pop();
            }
            
            if (currentWord) {
                Renderer.displayCurrentWord();
            } else {
                elements.wordDisplay.textContent = "没有可用单词，请先添加单词到当前题库";
                elements.wordPos.textContent = "";
                elements.exampleSentence.textContent = "";
                elements.playSoundBtn.disabled = true;
            }
        }

        // 检查答案
        function checkAnswer() {
            if (!currentWord || isAnswerChecked) return;
            
            const userAnswer = elements.userInput.value.trim().toLowerCase();
            let isCorrect = false;
            
            if (displayLang === 'cn') {
                // 检查英文拼写
                isCorrect = userAnswer === currentWord.en.toLowerCase();
            } else {
                // 检查中文释义
                const userAnswers = userAnswer.split(/[,，、;；]/).map(a => a.trim());
                const correctAnswers = currentWord.cn.map(c => c.trim().toLowerCase());
                
                // 只要有一个匹配就认为正确
                isCorrect = userAnswers.some(ans => correctAnswers.includes(ans));
            }
            
            // 更新结果显示
            if (isCorrect) {
                elements.result.textContent = `正确！答案是 ${displayLang === 'cn' ? currentWord.en : currentWord.cn.join('、')}`;
                elements.result.className = 'correct';
                correctCount++;
                
                // 如果在重练错题模式且答案正确，从错题中移除
                if (isRedoWrongMode) {
                    wrongWordList = wrongWordList.filter(word => word.id !== currentWord.id);
                    Renderer.renderWrongWords();
                }
                
                DataManager.updateWordStats(currentWord.id, true);
            } else {
                elements.result.textContent = `错误！正确答案是 ${displayLang === 'cn' ? currentWord.en : currentWord.cn.join('、')}`;
                elements.result.className = 'wrong';
                
                // 添加到错题本（去重）
                if (!wrongWordList.some(word => word.id === currentWord.id)) {
                    wrongWordList.push(currentWord);
                    Renderer.renderWrongWords();
                }
                
                DataManager.updateWordStats(currentWord.id, false);
            }
            
            totalCount++;
            updateStats();
            isAnswerChecked = true;
            DataManager.save();
        }

        // 播放单词发音
        function playWordSound(word) {
            if (!word) return;
            
            // 优先使用有道API发音
            const audioUrl = YoudaoAPI.getPronunciationUrl(word);
            if (audioUrl) {
                const audio = new Audio(audioUrl);
                audio.play().catch(error => {
                    console.error('播放有道发音失败，尝试浏览器语音合成:', error);
                    fallbackSpeechSynthesis(word);
                });
            } else {
                // 降级使用浏览器语音合成
                fallbackSpeechSynthesis(word);
            }
        }

        // 浏览器语音合成降级方案
        function fallbackSpeechSynthesis(word) {
            if ('speechSynthesis' in window) {
                const utterance = new SpeechSynthesisUtterance(word);
                utterance.lang = 'en-US';
                speechSynthesis.speak(utterance);
            } else {
                alert('您的浏览器不支持语音合成功能');
            }
        }

        // 从有道API获取单词信息并填充表单
        async function fetchAndFillWordInfo() {
            const word = elements.englishWord.value.trim();
            if (!word) {
                alert('请输入英文单词');
                return;
            }
            
            if (!YoudaoAPI.isConfigured()) {
                elements.apiError.style.display = "block";
                return;
            }
            
            // 显示加载状态
            elements.meaningLoading.style.display = "block";
            elements.meaningError.style.display = "none";
            elements.meaningOptionsContainer.style.display = "none";
            
            // 调用有道API获取单词信息
            const wordData = await YoudaoAPI.getWordDetail(word);
            
            // 隐藏加载状态
            elements.meaningLoading.style.display = "none";
            
            if (wordData && wordData.parts.length > 0) {
                // 填充音标
                elements.phonetic.value = wordData.phonetic || '';
                
                // 显示含义选项
                meaningSuggestions = wordData.parts;
                elements.meaningOptions.innerHTML = '';
                
                wordData.parts.forEach((part, index) => {
                    const div = document.createElement('div');
                    div.className = 'meaning-option';
                    div.innerHTML = `<strong>${part.pos}</strong> ${part.meaning}`;
                    div.addEventListener('click', () => {
                        // 选中该含义
                        elements.partOfSpeech.value = part.pos || '——   词组';
                        elements.chineseMeaning.value = part.meaning;
                        
                        // 高亮选中项
                        document.querySelectorAll('.meaning-option').forEach(el => {
                            el.style.backgroundColor = '#f0f7ff';
                        });
                        div.style.backgroundColor = '#e1f0fa';
                    });
                    elements.meaningOptions.appendChild(div);
                });
                
                elements.meaningOptionsContainer.style.display = "block";
                
                // 自动填充第一个含义
                if (wordData.parts[0]) {
                    elements.partOfSpeech.value = wordData.parts[0].pos || '——   词组';
                    elements.chineseMeaning.value = wordData.parts[0].meaning;
                    if (elements.meaningOptions.firstChild) {
                        elements.meaningOptions.firstChild.style.backgroundColor = '#e1f0fa';
                    }
                }
                
                // 填充例句
                if (wordData.examples && wordData.examples.length > 0) {
                    elements.exampleSentenceInput.value = wordData.examples[0].en || '';
                }
                
                // 启用发音按钮
                elements.pronounceNewWord.disabled = false;
            } else {
                elements.meaningError.style.display = "block";
                elements.pronounceNewWord.disabled = false; // 即使没有释义也允许发音
            }
        }

        // 保存API设置
        function saveApiSettings() {
            const appKey = elements.youdaoAppKey.value.trim();
            const appSecret = elements.youdaoAppSecret.value.trim();
            
            if (!appKey || !appSecret) {
                alert('请输入完整的应用ID和密钥');
                return;
            }
            
            youdaoConfig.appKey = appKey;
            youdaoConfig.appSecret = appSecret;
            
            localStorage.setItem('youdaoApiConfig', JSON.stringify(youdaoConfig));
            alert('API设置已保存');
            elements.apiError.style.display = "none";
        }

        // 加载API设置
        function loadApiSettings() {
            const savedConfig = localStorage.getItem('youdaoApiConfig');
            if (savedConfig) {
                try {
                    youdaoConfig = JSON.parse(savedConfig);
                    elements.youdaoAppKey.value = youdaoConfig.appKey || '';
                    elements.youdaoAppSecret.value = youdaoConfig.appSecret || '';
                } catch (e) {
                    console.error('加载API配置失败:', e);
                }
            }
        }

        // 设置事件监听器
        function setupEventListeners() {
            // 标签页切换
            elements.tabBtns.forEach(btn => {
                btn.addEventListener('click', () => {
                    const tabId = btn.getAttribute('data-tab');
                    
                    // 更新按钮状态
                    elements.tabBtns.forEach(b => b.classList.remove('active'));
                    btn.classList.add('active');
                    
                    // 更新内容显示
                    elements.tabContents.forEach(content => {
                        content.classList.remove('active');
                    });
                    document.getElementById(tabId).classList.add('active');
                });
            });
            
            // 网络状态变化监听
            window.addEventListener('online', updateNetworkStatus);
            window.addEventListener('offline', updateNetworkStatus);
            
            // 练习页面事件
            elements.checkBtn.addEventListener('click', checkAnswer);
            elements.nextBtn.addEventListener('click', getNextWord);
            elements.playSoundBtn.addEventListener('click', () => {
                if (currentWord) {
                    playWordSound(currentWord.en);
                }
            });
            
            // 输入框回车事件
            elements.userInput.addEventListener('keydown', (e) => {
                if (e.key === 'Enter') {
                    if (isAnswerChecked) {
                        getNextWord();
                    } else {
                        checkAnswer();
                    }
                }
            });
            
            // 显示语言切换
            elements.displayLang.addEventListener('change', (e) => {
                displayLang = e.target.value;
                if (currentWord) {
                    Renderer.displayCurrentWord();
                }
            });
            
            // 题库切换
            elements.librarySelector.addEventListener('change', (e) => {
                currentLibraryId = e.target.value;
                elements.wordLibrary.value = currentLibraryId;
                initWordPool();
                getNextWord();
            });
            
            elements.wordLibrary.addEventListener('change', (e) => {
                currentLibraryId = e.target.value;
                elements.librarySelector.value = currentLibraryId;
                Renderer.renderWordList();
                updateWordCount();
                initWordPool();
            });
            
            // 错题本事件
            elements.redoWrongBtn.addEventListener('click', () => {
                if (wrongWordList.length === 0) {
                    alert('没有错题可以练习');
                    return;
                }
                
                isRedoWrongMode = true;
                getNextWord();
                // 切换到练习标签
                elements.tabBtns.forEach(b => b.classList.remove('active'));
                document.querySelector('.tab-btn[data-tab="practice"]').classList.add('active');
                elements.tabContents.forEach(content => content.classList.remove('active'));
                document.getElementById('practice').classList.add('active');
                
                elements.tip.textContent = '正在重练错题模式 - 第一次回车确认答案，第二次回车下一题';
            });
            
            elements.clearWrongBtn.addEventListener('click', () => {
                if (confirm('确定要清空所有错题记录吗？')) {
                    wrongWordList = [];
                    DataManager.save();
                    Renderer.renderWrongWords();
                }
            });
            
            // 单词管理事件
            elements.createLibrary.addEventListener('click', () => {
                const libraryName = prompt('请输入新题库名称:');
                if (libraryName && libraryName.trim()) {
                    const libraryId = 'lib_' + Date.now();
                    wordLibraries[libraryId] = {
                        name: libraryName.trim(),
                        words: []
                    };
                    currentLibraryId = libraryId;
                    DataManager.save();
                    Renderer.renderLibrarySelectors();
                    Renderer.renderWordList();
                    updateWordCount();
                }
            });
            
            elements.renameLibrary.addEventListener('click', () => {
                if (currentLibraryId === 'default') {
                    alert('默认题库不能重命名');
                    return;
                }
                
                const newName = prompt('请输入新的题库名称:', wordLibraries[currentLibraryId].name);
                if (newName && newName.trim() && newName.trim() !== wordLibraries[currentLibraryId].name) {
                    wordLibraries[currentLibraryId].name = newName.trim();
                    DataManager.save();
                    Renderer.renderLibrarySelectors();
                }
            });
            
            elements.deleteLibrary.addEventListener('click', () => {
                if (currentLibraryId === 'default') {
                    alert('默认题库不能删除');
                    return;
                }
                
                if (confirm(`确定要删除"${wordLibraries[currentLibraryId].name}"吗？所有单词将被永久删除。`)) {
                    delete wordLibraries[currentLibraryId];
                    currentLibraryId = 'default';
                    DataManager.save();
                    Renderer.renderLibrarySelectors();
                    Renderer.renderWordList();
                    updateWordCount();
                }
            });
            
            // 单词表单提交
            elements.wordForm.addEventListener('submit', (e) => {
                e.preventDefault();
                
                const enWord = elements.englishWord.value.trim();
                const pos = elements.partOfSpeech.value;
                const cnMeaning = elements.chineseMeaning.value.trim();
                const phonetic = elements.phonetic.value.trim();
                const example = elements.exampleSentenceInput.value.trim();
                
                if (!enWord || !cnMeaning) {
                    alert('请填写英文单词和中文释义');
                    return;
                }
                
                // 创建新单词对象
                const newWord = {
                    en: enWord,
                    phonetic: phonetic,
                    pos: pos,
                    cn: cnMeaning.split(/[,，、;；]/).map(m => m.trim()),
                    examples: example ? [example] : [],
                    id: `${Date.now()}-${currentLibraryId}`,
                    errorCount: 0,
                    correctCount: 0
                };
                
                // 添加到题库
                DataManager.addWord(newWord);
                
                // 刷新列表
                Renderer.renderWordList();
                updateWordCount();
                
                // 重置表单
                elements.wordForm.reset();
                elements.meaningOptionsContainer.style.display = "none";
                elements.pronounceNewWord.disabled = true;
                
                alert('单词添加成功');
            });
            
            // 生成例句按钮
            elements.generateExample.addEventListener('click', async () => {
                const word = elements.englishWord.value.trim();
                if (!word) {
                    alert('请先输入英文单词');
                    return;
                }
                
                if (!YoudaoAPI.isConfigured()) {
                    elements.apiError.style.display = "block";
                    return;
                }
                
                elements.exampleError.style.display = "none";
                
                try {
                    elements.generateExample.disabled = true;
                    elements.generateExample.textContent = "生成中...";
                    
                    // 使用有道API获取例句
                    const wordData = await YoudaoAPI.getWordDetail(word);
                    
                    if (wordData && wordData.examples && wordData.examples.length > 0) {
                        elements.exampleSentenceInput.value = wordData.examples[0].en;
                    } else {
                        elements.exampleError.textContent = "未能获取到例句，请手动输入";
                        elements.exampleError.style.display = "block";
                    }
                } catch (error) {
                    console.error('生成例句失败:', error);
                    elements.exampleError.textContent = "生成例句失败，请重试";
                    elements.exampleError.style.display = "block";
                } finally {
                    elements.generateExample.disabled = false;
                    elements.generateExample.textContent = "生成";
                }
            });
            
            // 发音按钮事件
            elements.pronounceNewWord.addEventListener('click', () => {
                const word = elements.englishWord.value.trim();
                if (word) {
                    playWordSound(word);
                }
            });
            
            // 获取单词信息按钮
            elements.fetchWordInfo.addEventListener('click', fetchAndFillWordInfo);
            
            // API设置保存按钮
            elements.saveApiSettings.addEventListener('click', saveApiSettings);
        }

        // 页面加载完成后初始化

            // 加载CryptoJS用于生成有道API签名
            const script = document.createElement('script');
            script.src = 'https://cdn.jsdelivr.net/npm/crypto-js@4.1.1/crypto-js.min.js';
            script.onload = init;
            document.head.appendChild(script);
        });
    </script>
